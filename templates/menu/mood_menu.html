@admin.register(Food) class FoodAdmin(admin.ModelAdmin): # 'mood_tag' এবং 'is_surprise_item' লিস্টে দেখার জন্য যোগ করা হয়েছে list_display = ('display_image', 'name', 'category', 'mood_tag', 'price_badge', 'is_surprise_item', 'is_available') # সাইডবারে ফিল্টার করার জন্য 'mood_tag' যোগ করা হয়েছে list_filter = ('category', 'is_available', 'mood_tag', 'is_surprise_item') search_fields = ('name',) # লিস্ট থেকেই যেন মুড এবং সারপ্রাইজ স্ট্যাটাস চেঞ্জ করা যায় list_editable = ('is_available', 'mood_tag', 'is_surprise_item') def display_image(self, obj): if obj.image: return format_html('', obj.image.url) return format_html('No Image') display_image.short_description = "Preview" def price_badge(self, obj): return format_html('৳{}', obj.price) price_badge.short_description = "Price"
